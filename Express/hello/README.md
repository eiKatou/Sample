http://localhost:3000

    $ express hello --view=ejs

      create : hello/routes/users.js
      create : hello/views/
      create : hello/views/error.ejs
      create : hello/views/index.ejs
      create : hello/app.js
      create : hello/package.json
      create : hello/bin/
      create : hello/bin/www

      change directory:
        $ cd hello

      install dependencies:
        $ npm install

      run the app:
        $ DEBUG=hello:* npm start
